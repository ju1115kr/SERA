kr_construction_orders = pd.read_excel("Data/건설/건설수주액경상 월별.xlsx")

# '수주총액 계' 데이터 추출 (첫 번째 행)
total_order_value = kr_construction_orders.iloc[0, 2:]

# 데이터 프레임으로 변환
monthly_kr_construction_orders = pd.DataFrame({
    '월': total_order_value.index,
    'KR_construction_orders': total_order_value.values
})

# 월별 데이터를 datetime 형식으로 변환
monthly_kr_construction_orders['월'] = pd.to_datetime(monthly_kr_construction_orders['월'], format='%Y.%m')

# Year와 Month 컬럼을 분리
monthly_kr_construction_orders['Year'] = monthly_kr_construction_orders['월'].dt.year
monthly_kr_construction_orders['Month'] = monthly_kr_construction_orders['월'].dt.month

# 컬럼 순서 변경
monthly_kr_construction_orders = monthly_kr_construction_orders[['Year', 'Month', 'KR_construction_orders']]

# 데이터 확인
monthly_kr_construction_orders
